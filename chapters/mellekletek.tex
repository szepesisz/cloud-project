services:
  portainer:
    container_name: portainer
    image: portainer/portainer-ce:${PORTAINER_VERSION:-latest}

    restart: always
    ports:
      - 8100:9000

      - $DATA_DIR/$COMPOSE_PROJECT_NAME/portainer/data:/data
    networks:
      - infranet

  gitea:
    image: gitea/gitea:${GITEA_VERSION:-latest}
    container_name: gitea
    environment:
      USER_UID: 1000
      USER_GID: 1000
      GITEA__WEBHOOK__ALLOWED_HOST_LIST: "drone"
      GITEA__SERVER__ROOT_URL: "http://localhost:8101/"
    volumes:
      - $DATA_DIR/$COMPOSE_PROJECT_NAME/gitea/data:/data
    ports:
      - 8101:3000
      - 8122:22
    networks:
      - infranet

  drone:
    image: drone/drone:${DRONE_VERSION:-latest}
    container_name: drone
    environment:
      # DRONE_GITEA_SERVER: http://localhost:8101
      DRONE_GITEA_SERVER: http://gitea:3000
      DRONE_GITEA_CLIENT_ID: ${DRONE_GITEA_CLIENT_ID}
      DRONE_GITEA_CLIENT_SECRET: ${DRONE_GITEA_CLIENT_SECRET}

      DRONE_SERVER_PROTO: http

      - $DATA_DIR/$COMPOSE_PROJECT_NAME/drone/data:/data

      - "8102:80"

      - gitea
    networks:
      - infranet

    image: drone/drone-runner-docker:${DRONE_DOCKER_RUNNER_VERSION:-latest}

      DRONE_RPC_PROTO: http

    volumes:
      - /var/run/docker.sock:/var/run/docker.sock
    depends_on:
      - drone
    networks:
      - infranet

networks:
  infranet:

\subsection{A demo service}


\subsubsection*{Dockerfile}
\begin{lstlisting}
  ARG PYTHON_VERSION=3.13

FROM python:${PYTHON_VERSION}-slim AS base
ENV PIP_ROOT_USER_ACTION=ignore
RUN apt-get update && \
    yes | apt-get install git && \
    pip install build
COPY . .
RUN python -m build .

FROM python:${PYTHON_VERSION}-slim AS wheel
ENV PIP_ROOT_USER_ACTION=ignore
COPY --from=base dist/*.whl .


FROM wheel AS build
RUN pip install $(find ./*.whl)  && \
    rm $(find ./*.whl) && \
    mkdir /app
WORKDIR /app
ENV PYTHONBUFFERED=1
ENTRYPOINT ["app"]


FROM build AS test
WORKDIR /
COPY --from=base . .
RUN python \
        -m pip install \
            -r requirements_test.txt  \
            pytest pytest-cov \
            bandit ruff \
    && mkdir \
        -p testresults \
    && python \
        -m pytest \
            --junitxml testresults/unittests.junit.xml \
            --cov-report xml:testresults/coverage.xml \
            --color=no \
        || true \
    && python \
        -m bandit \
            --exit-zero \
            -r \
            -f xml \
            src \
        > testresults/bandit.junit.xml \
    && sed \
        -i 's/classname\=\"src/classname\=\"bandit\.src/g' \
        testresults/bandit.junit.xml \
    && python \
        -m ruff check \
            --exit-zero \
            src \
        > testresults/ruff.junit.xml

\end{lstlisting}

\subsubsection*{.drone.yml}
\begin{lstlisting}
kind: pipeline
type: docker
name: default

clone:
  disable: true

steps:
- name: clone
  image: alpine/git
  commands:
  - git clone http://192.168.0.173:8101/oe-lorawan/oe-lorawan-sniffer.git .
  - git fetch --tags
  - git checkout $DRONE_COMMIT

- name: test
  image: docker:24.0
  privileged: true
  commands:
  - docker build --target test .

  volumes:
    - name: docker
      path: /var/run/docker.sock


- name: build
  image: docker:24.0
  privileged: true
  commands:
  - docker build --tag oe-lorawan/oe-lorawan-sniffer --target build .
  - docker tag oe-lorawan/oe-lorawan-sniffer oe-lorawan/oe-lorawan-sniffer:$(docker run oe-lorawan/oe-lorawan-sniffer --version)
  - docker tag oe-lorawan/oe-lorawan-sniffer oe-lorawan/oe-lorawan-sniffer:latest

  volumes:
    - name: docker
      path: /var/run/docker.sock

- name: deploy
  image: curlimages/curl:8.17.0
  secrets: [ portainer_api_key ]
  environment:
    PORTAINER_API_KEY:
      from_secret: portainer_api_key

  commands:
  - echo "hi $PORTAINER_API_KEY"
  - "curl -i -X PUT
     http://192.168.0.173:8100/api/stacks/2/git/redeploy?endpointId=1 -H
     \"X-API-KEY: $PORTAINER_API_KEY\" --json '{ \"ForceUpdate\": false }'"


volumes:
  - name: docker
    host:
      path: /var/run/docker.sock
\end{lstlisting}
